CREATE TABLE sessions (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  created_at timestamptz DEFAULT now(),
  bc_count int DEFAULT 0, bc_proces boolean DEFAULT false, bc_total int DEFAULT 0,
  ls_count int DEFAULT 0, ls_proces boolean DEFAULT false, ls_total int DEFAULT 0,
  priv_count int DEFAULT 0, priv_brut int DEFAULT 0, priv_tax int DEFAULT 0,
  priv_net int DEFAULT 0, grand_total int DEFAULT 0, note text
);
CREATE TABLE services (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  created_at timestamptz DEFAULT now(),
  name text NOT NULL, category text, price int DEFAULT 0,
  taxable boolean DEFAULT false, visible boolean DEFAULT true,
  tag text, description text
);
